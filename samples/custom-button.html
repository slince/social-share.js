一、定制分享内容和链接
            
                new SocialShare('.social-share-container1', {

                    // 定义通用的分享参数
                    title: '百度搜索',
                    url: 'http://www.baidu.com',
                    summary: '一个中文搜索网站',
                    image: 'https://ss0.bdstatic.com/5aV1bjqh_Q23odCf/static/superman/img/logo/bd_logo1_31bdc765.png',

                    // 单独定义微博的分享参数,其它的同
                    weibo: {
                        title: '网易',
                        url: 'http://www.qq.com',
                        summary: '网易门户',
                        image: 'https://ss0.bdstatic.com/5aV1bjqh_Q23odCf/static/superman/img/logo/bd_logo1_31bdc765.png',
                    }
                });
